
#include <stdio.h>
#include <string.h>
struct Aluno {
    int nota1;
    int nota2;
    
}
int main() {
    
    struct Aluno tNotas; //registra num espaço de memoria variavel do tipo Aluno com o nome tnotas
    
    tNotas.nota1 = 2; //adiciona dentro do espaço de memoria do tnotas, o valor do nota1.
    tNotas.nota2 = 3;
    

printf("Digite a 1° nota do aluno: %d\n", tNotas.nota1);
printf("Digite a 2° nota do aluno: %d\n", tNotas.nota2);

scanf("%d", &nota1);
scanf("%d", &nota2);
float media = (nota1+nota2)/2;
printf("Professor digitou: %d\n", nota1);
printf("Professor digitou: %d\n", nota2);
printf("A media de notas do aluno é %f\n", media);    
printf("%f", &media);


return 0;
}
